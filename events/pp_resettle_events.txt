namespace = PP

# reroute to pop
# This = Planet scope
# From = Country
# FromFrom = Pop
planet_event = {
	id = PP.10
    hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = owner
		
		OR = {
			has_modifier = planet_population_control_gestalt
			has_modifier = planet_robot_assembly_control
		}

		# don't decolonize planet
        planet = { num_pops > 1 }
	}
	
	immediate = {
        log = "PP::PP.10(THIS = [This.GetName], FROM = [From.GetName], FROMFROM = [From.From.GetName])"

		owner = {
			random_owned_planet = {
				limit = {
					NOR = {
						has_modifier = planet_population_control_gestalt
						has_modifier = planet_robot_assembly_control
					}

					# only colonized planets
					is_colony = yes
					is_under_colonization = no 

					# not under siege
					has_orbital_bombardment = no
				}

				log = "PP::'Relocating [From.From.GetName] to [This.GetClassName] Planet [This.GetName]!'"

				resettle_pop = { pop = fromfrom planet = this }
			}
		}
	}
}

# relocate abductee
# This = Pop scope
# From = planet abducted from
pop_event = {
    id = PP.20
    hide_window = yes
    is_triggered_only = yes

    trigger = {
		exists = owner

		# don't decolonize planet
        planet = { num_pops > 1 }

        # only relocate pops that are not already on a purge world
        NOT = { planet = { has_modifier = pp_purge_world } }
    }
    
	immediate = {
        log = "PP::PP.20(THIS = [This.GetName], FROM = [From.GetName], FROMFROM = [From.From.GetName])"

		owner = {
			random_owned_planet = {
				limit = {
					has_modifier = pp_purge_world

					# only colonized planets
					is_colony = yes
					is_under_colonization = no 

					# not under siege
					has_orbital_bombardment = no
				}

				log = "PP::'Relocating [Root.GetName] to [This.GetClassName] Planet [This.GetName]!'"

				resettle_pop = { pop = root planet = this }
			}
		}
    }
}
